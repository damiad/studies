#include <stdio.h>
#include <stdlib.h>
#include <limits.h>

#define MAX_NODES 1000

int graph[MAX_NODES][MAX_NODES];
int distances[MAX_NODES];
int visited[MAX_NODES];
int num_nodes;

void dijkstra(int start_node, int end_node) {
    int i, j;
    
    for (i = 0; i < num_nodes; i++) {
        distances[i] = INT_MAX;
        visited[i] = 0;
    }
    
    distances[start_node] = 0;
    
    for (i = 0; i < num_nodes - 1; i++) {
        int min_distance = INT_MAX;
        int min_index = -1;
        
        for (j = 0; j < num_nodes; j++) {
            if (!visited[j] && distances[j] < min_distance) {
                min_distance = distances[j];
                min_index = j;
            }
        }
        
        if (min_index == -1) {
            break;
        }
        
        visited[min_index] = 1;
        
        for (j = 0; j < num_nodes; j++) {
            if (graph[min_index][j] != 0 && !visited[j]) {
                int dist = distances[min_index] + graph[min_index][j];
                
                if (dist < distances[j]) {
                    distances[j] = dist;
                }
            }
        }
    }
    
    if (distances[end_node] != INT_MAX) {
        printf("Shortest path between nodes %d and %d is %d\n",
           start_node, end_node, distances[end_node]);
    } else {
        printf("No path between nodes %d and %d\n", start_node, end_node);
    }
}

int main() {
    int i, j;
    int start_node, end_node;
    
    printf("Enter the number of nodes in the graph (max %d): ", MAX_NODES);
    scanf("%d", &num_nodes);
    
    printf("Enter the adjacency matrix for the graph:\n");
    
    for (i = 0; i < num_nodes; i++) {
        for (j = 0; j < num_nodes; j++) {
            scanf("%d", &graph[i][j]);
        }
    }
    
    printf("Enter the start node: ");
    scanf("%d", &start_node);
    
    printf("Enter the end node: ");
    scanf("%d", &end_node);
    
    dijkstra(start_node, end_node);
    
    return 0;
}
